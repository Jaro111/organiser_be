try {
    // Create a new task object with the owner set to the current user
    const newTask = new Task({
      title,
      description,
      owner: userId,
      invitedUsers: invitedUsers
        ? invitedUsers.map((inviteeId) => ({
            user: inviteeId,
            accepted: false // Set all invited users to 'not yet accepted'
          }))
        : []
    });

    // Save the new task to the database
    await newTask.save();

    return res.status(201).json({ message: 'Task created successfully', task: newTask });
  } catch (error) {
    return res.status(500).json({ message: error.message });
  }
};
